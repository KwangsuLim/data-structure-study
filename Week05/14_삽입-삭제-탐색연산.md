## 이동연산

```python
def moveAfter(self, a, x):
    # 노드 a를 노드 x 다음으로 이동
    splice(a,a,x)
```

```python
def moveBefore(a, x):
    # a를 x전으로 이동
    splice(a,a,x.prev)
```

## 탐색 연산

```python
def search(self,key):
    v = self.head #dummy node
    while v.next != self.head: #한바퀴 돌기
        if v.key == key:
            return v
        v = v.next
    return None
```

## 삭제 연산
```python
def remove(x):
    if x == None or x == self.head:
        return 
    x.prev.next = x.next
    x.next.prev = x.prev
    del x
```

```python
def popFront(self):
    # 리스트 비어있으면 종료
    if self.head is None:
        return
    
    temp = self.head

    self.head = self.head.next # head를 다음 노드로 옮김

    if self.head is not None: # 새로운 head가 None이 아니라면 prev를 None으로
        self.head.prev = None
    del temp #기존 노드 삭제
```

```python
def popBack(self):
    #리스트 비어있으면 종료
    if self.tail is None:
        return
    temp = self.tail

    if self.head == self.tail:
        self.head = self.tail = None
    else:
        self.tail = self.tail.prev # tail을 이전 노드로 이동
        self.tail.next = None # 새로운 tail의 next를 None으로 지정
    del temp
```
