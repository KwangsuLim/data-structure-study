```python
class Node:
    def __init__(self, key=None):
        self.key = key
        self.next = None
    def __str__(self):
        return str(self.key)


class SinglyLinkedList:
    def __init__(self):
        self.head = None
        self.size = 0

    def __len__(self):
        return self.size

    def __str__(self):
        vals = []
        cur = self.head
        while cur:
            vals.append(str(cur.key))
            cur = cur.next
        return " -> ".join(vals) if vals else "∅"

    # 맨 앞 삽입: O(1)
    def pushFront(self, key):
        new_node = Node(key)
        new_node.next = self.head
        self.head = new_node
        self.size += 1

    # 맨 뒤 삽입: O(n)
    def pushBack(self, key):
        v = Node(key)
        if self.head is None:
            self.head = v
        else:
            tail = self.head
            while tail.next:
                tail = tail.next
            tail.next = v
        self.size += 1

    # 맨 앞 삭제: O(1)
    def popFront(self):
        if self.head is None:
            return None
        x = self.head
        key = x.key
        self.head = x.next
        self.size -= 1
        del x
        return key

    # 맨 뒤 삭제: O(n)
    def popBack(self):
        if self.head is None:
            return None

        # 노드가 하나뿐인 경우
        if self.head.next is None:
            key = self.head.key
            del self.head
            self.head = None
            self.size -= 1
            return key

        # 노드가 두 개 이상인 경우: prev, tail 러닝 포인터
        prev, tail = None, self.head
        while tail.next:
            prev = tail
            tail = tail.next

        # tail은 마지막 노드, prev는 tail 직전
        key = tail.key
        prev.next = None
        del tail
        self.size -= 1
        return key
```
<br>

- PushFront: $O(1)$ 
- PopFront: $O(1)$
- pushBack: $O(n)$
- popback: $O(n)$
